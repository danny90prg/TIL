# Problem
https://www.hackerrank.com/challenges/the-birthday-bar/problem

the birthday bar

# Solve
```c++
int birthday(vector<int> s, int d, int m) {
    int n = s.size();
    int sum[105];
    int count = 0;
    sum[0] = 0;
    for (int i = 0; i < n; i++)
        sum[i + 1] = sum[i] + s[i];
    for (int i = 0; i <= n - m; i++)
        if (sum[i + m] - sum[i] == d)   count++;
    return count;
}
```
