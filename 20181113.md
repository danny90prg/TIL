int equalizeArray(vector<int> arr) {
    int cnt[101] = {0,};
    int n = arr.size();
    int max = 0;
    for(int i = 0; i<n; i++){
        cnt[arr[i]]++;
        if(cnt[arr[i]] > cnt[max])  max = arr[i];
    }
    return n - cnt[max];
}
